62/** \brief Register Map
63 *
64 * ## Register Map
65 *
66 * |Register Number | Description | r/w |
67 * | -------------- |------------ | --- |
68 * | 0 | PORTA input data register, bit 5 is LED | r |
69 * | 1 | PORTA output data register, bit 5 is LED | r, bit 5 is w |
70 * | 2 | optionFlags: see below | r/w |
71 * | 3 | DAC Channel 1 output val | r, bottom 12 bits w |
72 * | 4 | Reserved for DAC Channel 2, but not implemented | r |
73 * | 5 | Current millisecond_timer value | r |
74 * | 6 | Period between ADC samples in ms * | r/w |
75 * | 7 | Number of times ADC overrun flag has been set | r |
76 * | 8 | Time between DAC conversion in 100 us * | r/w 16 bits |
77 * | 9 | DAC waveform & waveform amplitude sel. * | r, w bits 6-10 |
78 *
79 * ### Option flags
80 *
81 * |Big Number | Description |
82 * | -------------- |------------ |
83 * | 0 | if 0: stream ADC, if 1: stream counter |
84 * | 1 | if 1: disable & re-enable DAC, then reset bit to 0, if 0: nothing |
85 *
86 * Register 6: ADC sample period only written to ADC register when 'n' command
87 * received
88 *
89 * Register 8: default 100, so update every 10 ms
90 *
91 * ### Waveform generation with register 9
92 *
93 * Register 9: default 0. Bits 6-7 (start from 0) control waveform generation.
94 * 00: disabled, 01: noise, 1x: triangle. Bits 8-11 are amplitude: where the
95 * max-min is 2^(n+1)-1, where n is the value (at least for triangle), maxing
96 * out at 0b1011=0xB. The triangle starts from reg 3 and goes up to the max-min
97 * value programmed, then comes back down. The noise is centered around
98 * register 3.
99 *
100 * So, write to the register: 0xYZ0, where Y is the amplitude: 0-B, and Z is
101 * the waveform type: 0 for none, 1 for nosie, 8 for triangle.
102 *
103 * I'd say the noise has an RMS in the region of 10 counts. The mask doesn't
104 * change the RMS that much.
105 *
106 * **It may be necessary to reset the DAC (see option flags) to change the
107 * amplitude and/or disable waveform generation**
108 *
109 * Register 9 examples:
110 *
111 * - triangle with max-min = 127: 0b10110000000 = 0x680
112 * - full scale triangle: write 9 0xB80, write 3 0, write 8 3
113 * - noise with large amplitude: 0xB10
114 * - flat, waveform disabled: 0
115 *
116 * @see register_write_masks
117 *
118 */

163static void adc_setup(void) {
164 // With nChans == 1 and ADC_MODE_SCAN, there is only one conversion per
165 // "trigger" You can manually trigger by setting ADSTART in ADC_CR
166 // ADSTART will stay set and only be cleared when EOSEQ is set
167 // After the conversion is complete EOC and EOSEQ are both set (in ADC_ISR)
168 // You can read the data from ADC_DR
169 rcc_periph_clock_enable(RCC_ADC);
170 rcc_periph_clock_enable(RCC_GPIOA);
171
172 // Only pins A0=ADC_IN0 happens to also be Arduino A0
173 gpio_mode_setup(GPIOA, GPIO_MODE_ANALOG, GPIO_PUPD_NONE, GPIO0);
174#define nChans 1
175 uint8_t adc_channel_sequence[nChans] = {0};
176
177 adc_power_off(ADC);
178 adc_set_clk_source(ADC, ADC_CLKSOURCE_ADC);
179 adc_calibrate(ADC);
180 adc_set_operation_mode(
181 ADC, ADC_MODE_SCAN); // ADC_MODE_SCAN is ~cont and ~discon;
182 // ADC_MODE_SCAN_INFINITE is cont and ~dicon
183 adc_disable_external_trigger_regular(ADC);
184 adc_set_right_aligned(ADC);
185 adc_enable_temperature_sensor();
186 adc_set_sample_time_on_all_channels(ADC, ADC_SMPTIME_071DOT5);
187 adc_set_regular_sequence(ADC, nChans, adc_channel_sequence);
188 adc_set_resolution(ADC, ADC_RESOLUTION_12BIT);
189 adc_disable_analog_watchdog(ADC);
190 adc_power_on(ADC); // this sync waits until ADRDY is set
191}

193static void dac_setup(void) {
194 // Setup TIM6
195 rcc_periph_clock_enable(RCC_TIM6);
196 TIM_PSC(TIM6) = rcc_ahb_frequency / 100 - 1; // tick every 100 us
197 TIM_ARR(TIM6) = 100; // by default, update every 10 ms
198 TIM_CR2(TIM6) |= TIM_CR2_MMS_UPDATE; // trigger output on update
199 TIM_CR1(TIM6) |= TIM_CR1_ARPE; // wait for update to update the value of
200 // auto-reload shadow reg
201 TIM_CR1(TIM6) |= TIM_CR1_CEN; // enable
202
203 // Setting this up to trigger using timer 6, so setting that up here too.
204 rcc_periph_clock_enable(RCC_DAC);
205 rcc_periph_clock_enable(RCC_GPIOA);
206
207 // Pin A4 is DAC1 output and Arduino A2 on board
208 // Pin A5 is DAC2 output and Arduino D13 on board
209 // Only enable DAC1 for now
210 gpio_mode_setup(GPIOA, GPIO_MODE_ANALOG, GPIO_PUPD_NONE, GPIO4);
211
212 dac_disable(DAC1, DAC_CHANNEL1);
213 dac_buffer_enable(DAC1, DAC_CHANNEL1);
214 dac_disable_waveform_generation(DAC1, DAC_CHANNEL1);
215 dac_set_trigger_source(DAC1, DAC_CR_TSEL1_T6);
216 dac_trigger_enable(DAC1, DAC_CHANNEL1);
217 dac_enable(DAC1, DAC_CHANNEL1);
218}

221int main(void) {
222
223 Try {
224 SETUP_ASC_DEVICE_W_REGISTER_POINTERS(register_map, register_write_masks,
225 nRegs);
226
227 millisec_timer_systick_setup(rcc_ahb_frequency);
228 gpio_setup();
229 adc_setup();
230 dac_setup();
231
232 // USART2 TX: PA2 AF1
233 // USART2 RX: PA3 AF1
234 rcc_periph_clock_enable(RCC_GPIOA);
235 rcc_periph_clock_enable(RCC_USART2);
236 setup_usart(ASC_USART, 9600, GPIOA, GPIO2, GPIO_AF1, GPIOA, GPIO3,
237 GPIO_AF1);
238 nvic_enable_irq(NVIC_USART2_IRQ);
239 usart_enable_rx_interrupt(ASC_USART);
240 usart_enable(ASC_USART);
241 }
242 Catch(e) { return e; }
243
244 while (1) {
245 Try {
246 const bool stream_state_before_receiving = streaming_is_on;
247 HANDLE_ASC_COMM_IN_POLLING_LOOP(ASC_USART);
248 // just started streaming so start this timer
249 if (streaming_is_on && !stream_state_before_receiving) {
250 millisec_timer_set_rel(&adc_timer, MILLISEC_TIMER_NOW,
251 adc_sample_period_ms);
252 }
253
254 if ((optionFlags >> 1) &
255 1) { // reset DAC, needed b/c it's hard to disable noise generation
256 dac_disable(DAC1, DAC_CHANNEL1);
257 optionFlags &= ~(1 << 1);
258 millisec_timer_set_rel(&dac_reset_timer, MILLISEC_TIMER_NOW,
259 DAC_RESET_TIME_MS);
260 }
261 if (millisec_timer_is_expired(&dac_reset_timer, MILLISEC_TIMER_NOW)) {
262 dac_enable(DAC1, DAC_CHANNEL1);
263 }
264
265 if (streaming_is_on && !(optionFlags & 1) &&
266 millisec_timer_is_expired_repeat(&adc_timer, MILLISEC_TIMER_NOW)) {
267 adc_start_conversion_regular(ADC);
268 }
269
270 if (READY_TO_STREAM_ASC_DEVICE_W_REGISTER_POINTERS) {
271 if (optionFlags & 1) { // counter stream mode
272 char counter_buffer[8];
273 convert_uint32_to_hex(counter, counter_buffer, true);
274 STREAM_TO_HOST_ASC_DEVICE_W_REGISTER_POINTERS(counter_buffer, 8);
275 counter++;
276 } else { // ADC stream mode
277 if (adc_eos(ADC)) {
278 if (adc_get_overrun_flag(ADC)) {
279 adc_n_overruns++;
280 adc_clear_overrun_flag(ADC);
281 }
282 static uint16_t adc_val;
283 static char adc_val_buffer[4];
284 adc_val = adc_read_regular(ADC);
285 ADC_ISR(ADC) |= (1 << ADC_ISR_EOSEQ) | (1 << ADC_ISR_EOC);
286 convert_uint16_to_hex(adc_val, adc_val_buffer, true);
287 STREAM_TO_HOST_ASC_DEVICE_W_REGISTER_POINTERS(adc_val_buffer + 1,
288 3);
289 }
290 }
291 }
292 }
293 Catch(e) { nExceptions++; }
294 }
295
296 return 0;
297}
