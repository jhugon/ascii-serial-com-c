73/** \brief Register Map
74 *
75 * ## Register Map
76 *
77 * |Register Number | Description | r/w |
78 * | -------------- |------------ | --- |
79 * | 0 | PORTA input data register, bit 5 is LED | r |
80 * | 1 | PORTA output data register, bit 5 is LED | r, bit 5 is w |
81 * | 2 | optionFlags: see below | r/w |
82 * | 3 | Current millisecond_timer value | r |
83 * | 4 | LED pulser prescaler | r/w 16 bits |
84 * | 5 | LED pulser period | r/w 16 bits |
85 * | 6 | LED pulser pulse length | r/w 16 bits |
86 * | 7 | Input capture pulse period | r |
87 * | 8 | Input capture pulse width | r |
88 * | 9 | Input capture overrun flags (bit 0 for period and 4 for width) | r |
89 *
90 * @see register_write_masks
91 *
92 */
93volatile REGTYPE *register_map[nRegs] = {
94 &GPIOA_IDR, // input data reg
95 &GPIOA_ODR, // output data reg
96 &optionFlags, // option flags
97 &MILLISEC_TIMER_NOW, // millisec timer value
98 &TIM_PSC(TIM_LED), // LED pulser prescaler
99 &TIM_ARR(TIM_LED), // LED pulser period
100 &TIM_CCR1(TIM_LED), // LED pulser pulse length
101 &ic_period, // Input capture pulse period
102 &ic_pulse_length, // Input capture pulse length
103 &ic_overrun, // Input capture overrun flags
104};

140 rcc_periph_clock_enable(RCC_USART2);

142 GPIO_AF1);
143 nvic_enable_irq(NVIC_USART2_IRQ);
144 usart_enable_rx_interrupt(ASC_USART);
145 usart_enable(ASC_USART);
146
147 rcc_periph_clock_enable(RCC_GPIO_LED);
148 rcc_periph_clock_enable(RCC_TIM_LED);
149 setup_timer_periodic_output_pulse(
150 TIM_LED, rcc_get_timer_clk_freq(TIM_LED) / pulser_tick_freq,
151 pulser_period, pulser_width, TIM_OC_LED, PORT_LED, PIN_LED, AF_TIM_LED);
152 timer_enable_counter(TIM_LED);
153
154 rcc_periph_clock_enable(IC_RCC_GPIO);
155 rcc_periph_clock_enable(IC_RCC_TIM);
156 setup_timer_capture_pwm_input(
157 IC_TIM, rcc_get_timer_clk_freq(IC_TIM) / pulser_tick_freq, 0xFFFF,
158 IC_TI, IC_PORT, IC_PIN, IC_AF);
159 timer_enable_counter(IC_TIM);
160 }
161 Catch(e) { return e; }
162
163 while (1) {
164 Try {
165 HANDLE_ASC_COMM_IN_POLLING_LOOP(ASC_USART);
166
167 if (timer_get_flag(IC_TIM, TIM_SR_CC1IF)) {
168 ic_pulse_length = TIM_CCR2(IC_TIM);
169 ic_period = TIM_CCR1(IC_TIM);
170 ic_overrun = 0;
171 if (timer_get_flag(IC_TIM, TIM_SR_CC1OF)) {
172 ic_overrun |= 1;
173 timer_clear_flag(IC_TIM, TIM_SR_CC1OF);
174 }
175 if (timer_get_flag(IC_TIM, TIM_SR_CC2OF)) {
176 ic_overrun |= 1 << 4;
177 timer_clear_flag(IC_TIM, TIM_SR_CC2OF);
178 }
179 }
180 }
181 Catch(e) { nExceptions++; }
182 }
183
184 return 0;
185}
